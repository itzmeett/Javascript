24. Write to check whether a number is negative, positive or zero?
<h3 id="answer"></h3>
<script>
alert('This program checks the number is POSITIVE, NEGATIVE or NEUTRAL')
const num = parseInt(prompt('Enter the Number  = '));
if (num > 0) {
    document.getElementById('answer').innerText = num + ' is POSITIVE NUMBER';
}
else if (num == 0) {
    document.getElementById('answer').innerText = num + ' is NEUTRAL NUMBER';
}
else if (num < 0) {
    document.getElementById('answer').innerText = num + ' is NEGATIVE NUMBER';
}   
else {
    document.getElementById('answer').innerHTML = 'You have not inserted any NUMBER';
}
</script>
