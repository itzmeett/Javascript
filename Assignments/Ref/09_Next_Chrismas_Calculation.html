9. Write a JavaScript program to calculate days left until next Christmas? 
<script>

let today = new Date; // it creates today object

let christmasYear = today.getFullYear(); // it gates current year

// Check if the current date is 
// already past by checking if the month 
// is December and the current day 
// is greater than 25 
if (today.getMonth() == 11 && today.getDate() > 25) {
    christmasYear += 1;
}

// set the christmasYear date of current year
let christmasYearRemainTime = new Date(christmasYear, 11, 25);

// get time in miliseconds
let time = 1000 * 60 * 60 * 24;

// calculation of getting remaining days
let remainDays = Math.ceil((christmasYearRemainTime.getTime() - today.getTime()) / time);

// put the answer on html element
console.log(remainDays);
document.getElementById('answer').innerHTML = remainDays +' days are remaining for Christmas-' + christmasYear;
</script>
