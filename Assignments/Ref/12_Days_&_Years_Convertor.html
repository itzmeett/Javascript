12. WAP to convert years into days and days into years? 
let numOfDays = prompt('Enter total days : ');
let numOfYear = prompt('Enter numbers of year : ');
let year, days;
function daysToYear(numOfDays) {
    year = parseInt(numOfDays / 365);  
    return year;
}

function yearsToDays(numOfYear) {
    days = (numOfYear * 365);
    return days;
}
document.getElementById('dd').innerHTML = numOfDays + ' days are approx = ' + daysToYear(numOfDays) + ' year';
document.getElementById('yy').innerHTML = numOfYear + ' year have approx = ' + yearsToDays(numOfYear) + ' days';
