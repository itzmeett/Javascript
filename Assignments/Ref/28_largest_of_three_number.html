28. Write to find the largest of three numbers in JS?
<h3 id="answer"></h3>
<script>
alert('This program checks the MAXIMUM NUMBER out of three numbers')
let num1 = parseInt(prompt('Enter Number - 1 : '));
let num2 = parseInt(prompt('Enter Number - 2 : '));
let num3 = parseInt(prompt('Enter Number - 3 : '));
if (num1 > num2 && num1 > num3) {
    document.getElementById("answer").innerText = num1 + ' is GRETEST NUMBER';
}
else if (num2 > num1 && num2 > num3) {
    document.getElementById("answer").innerText = num2 + ' is GRETEST NUMBER';
}
else if(num3 > num1 && num3 > num2){
    document.getElementById("answer").innerText = num3 + ' is GRETEST NUMBER';
}
else{
    document.getElementById("answer").innerText = 'You have not inserted any value';
}            
</script>
