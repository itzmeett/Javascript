25. Write to find number is even or odd using ternary operator in JS?
<h3 id="answer"></h3>
<script>
alert('This programs check the number is ODD or EVEN');
let num = parseInt(prompt('Enter any Number : '));

(num % 2 == 0) ? document.getElementById("answer").innerText = num + ' is EVEN' : (!num % 2 == 0) ? document.getElementById("answer").innerText = num + ' is ODD' :  document.getElementById("answer").innerText = 'You have not inserted any value';
</script>
