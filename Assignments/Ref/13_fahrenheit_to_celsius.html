13. Convert temperature Fahrenheit to Celsius? (Conditional logic Question) 
<script>            
function fahrenheitToCelsius(fahrenheit) {
    const celsius = (fahrenheit - 32) / 1.8;
    return celsius;
}
let fahrenheit = prompt('Enter temprature in fahrenheit : ');
document.getElementById('answer').innerHTML = fahrenheitToCelsius(fahrenheit);

</script>
