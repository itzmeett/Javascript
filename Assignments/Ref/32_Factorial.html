32. Write to print factorial of given number?
<script>
<h3 id="answer"></h3>
let num = prompt('Enter the number for factorial calculation = ');
let n = 1;
for (let index = 2; index <= num; index++) {
    n *= index;
    // return n;
}
// console.log(n);
document.getElementById('answer').innerText = 'Factorial of ' + num + ' = ' + n; 
</script>
